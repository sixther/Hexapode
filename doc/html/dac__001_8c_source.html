50 /* Initial DAC configuration. Maximum update rate is 1MHz; value to AOUT is 0 */
51 void IP_DAC_Init(IP_DAC_001_T *pDAC)
52 {
53  /* Set maximum update rate 1MHz */
54  IP_DAC_SetBias(pDAC, DAC_MAX_UPDATE_RATE_1MHz);
55 }

57 /* Update value to DAC buffer*/
58 void IP_DAC_UpdateValue(IP_DAC_001_T *pDAC, uint32_t dac_value)
59 {
60  uint32_t tmp;
61 
62  tmp = pDAC->CR & DAC_BIAS_EN;
63  tmp |= DAC_VALUE(dac_value);
64  /* Update value */
65  pDAC->CR = tmp;
66 }

68 /* Set Maximum update rate for DAC */
69 void IP_DAC_SetBias(IP_DAC_001_T *pDAC, uint32_t bias)
70 {
71  pDAC->CR &= ~DAC_BIAS_EN;
72 
73  if (bias == DAC_MAX_UPDATE_RATE_400kHz) {
74  pDAC->CR |= DAC_BIAS_EN;
75  }
76 }
