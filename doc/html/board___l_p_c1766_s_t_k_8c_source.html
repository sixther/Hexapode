167 /* Initialize debug output via UART for board */
168 void Board_Debug_Init(void)
169 {
170 #if defined(DEBUG_ENABLE)
171  Board_UART_Init(DEBUG_UART);
172 
173  /* Setup UART for 115.2K8N1 */
174  Chip_UART_Init(DEBUG_UART);
175  Chip_UART_SetBaud(DEBUG_UART, 115200);
176  Chip_UART_ConfigData(DEBUG_UART, UART_DATABIT_8, UART_PARITY_NONE, UART_STOPBIT_1);
177 
178  /* Enable UART Transmit */
179  Chip_UART_TxCmd(DEBUG_UART, ENABLE);
180 #endif
181 }

191 /* Gets a character from the UART, returns EOF if no character is ready */
192 int Board_UARTGetChar(void)
193 {
194 #if defined(DEBUG_ENABLE)
195  uint8_t data;
196 
197  if (Chip_UART_ReceiveByte(DEBUG_UART, &data) == SUCCESS) {
198  return (int) data;
199  }
200 #endif
201  return EOF;
202 }

282 uint32_t TOR_GetStatus(void)
283 {
284  uint8_t ret = NO_TOR_PRESSED;
285 
286  if (Chip_GPIO_ReadPortBit(LPC_GPIO, TOR0_GPIO_PORT_NUM, TOR0_GPIO_BIT_NUM) == 0x00) {
287  ret |= TOR_TOR0;
288  }
289  if (Chip_GPIO_ReadPortBit(LPC_GPIO, TOR1_GPIO_PORT_NUM, TOR1_GPIO_BIT_NUM) == 0x00) {
290  ret |= TOR_TOR1;
291  }
292  if (Chip_GPIO_ReadPortBit(LPC_GPIO, TOR2_GPIO_PORT_NUM, TOR2_GPIO_BIT_NUM) == 0x00) {
293  ret |= TOR_TOR2;
294  }
295  if (Chip_GPIO_ReadPortBit(LPC_GPIO, TOR3_GPIO_PORT_NUM, TOR3_GPIO_BIT_NUM) == 0x00) {
296  ret |= TOR_TOR3;
297  }
298  if (Chip_GPIO_ReadPortBit(LPC_GPIO, TOR4_GPIO_PORT_NUM, TOR4_GPIO_BIT_NUM) == 0x00) {
299  ret |= TOR_TOR4;
300  }
301  if (Chip_GPIO_ReadPortBit(LPC_GPIO, TOR5_GPIO_PORT_NUM, TOR5_GPIO_BIT_NUM) == 0x00) {
302  ret |= TOR_TOR5;
303  }
304 
305  return ret;
306 }

308 /* Sets the state of a board LED to on or off */
309 void Board_LED_Set(uint8_t LEDNumber, bool On)
310 {
311  /* There is only one LED */
312  if (LEDNumber == 0) {
313  Chip_GPIO_WritePortBit(LPC_GPIO, LED0_GPIO_PORT_NUM, LED0_GPIO_BIT_NUM, On);
314  }
315  if (LEDNumber == 1) {
316  Chip_GPIO_WritePortBit(LPC_GPIO, LED1_GPIO_PORT_NUM, LED1_GPIO_BIT_NUM, On);
317  }
318 }

328  }
329  return 0;
330 }
