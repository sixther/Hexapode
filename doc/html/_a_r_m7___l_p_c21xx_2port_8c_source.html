121  */
122 portSTACK_TYPE *pxPortInitialiseStack( portSTACK_TYPE *pxTopOfStack, pdTASK_CODE pxCode, void *pvParameters )
123 {
124 portSTACK_TYPE *pxOriginalTOS;
125 
126  /* Setup the initial stack of the task. The stack is set exactly as
127  expected by the portRESTORE_CONTEXT() macro.
128 
129  Remember where the top of the (simulated) stack is before we place
130  anything on it. */
131  pxOriginalTOS = pxTopOfStack;
132 
133  /* To ensure asserts in tasks.c don't fail, although in this case the assert
134  is not really required. */
135  pxTopOfStack--;
136 
137  /* First on the stack is the return address - which in this case is the
138  start of the task. The offset is added to make the return address appear
139  as it would within an IRQ ISR. */
140  *pxTopOfStack = ( portSTACK_TYPE ) pxCode + portINSTRUCTION_SIZE;
141  pxTopOfStack--;
142 
143  *pxTopOfStack = ( portSTACK_TYPE ) 0xaaaaaaaa; /* R14 */
144  pxTopOfStack--;
145  *pxTopOfStack = ( portSTACK_TYPE ) pxOriginalTOS; /* Stack used when task starts goes in R13. */
146  pxTopOfStack--;
147  *pxTopOfStack = ( portSTACK_TYPE ) 0x12121212; /* R12 */
148  pxTopOfStack--;
149  *pxTopOfStack = ( portSTACK_TYPE ) 0x11111111; /* R11 */
150  pxTopOfStack--;
151  *pxTopOfStack = ( portSTACK_TYPE ) 0x10101010; /* R10 */
152  pxTopOfStack--;
153  *pxTopOfStack = ( portSTACK_TYPE ) 0x09090909; /* R9 */
154  pxTopOfStack--;
155  *pxTopOfStack = ( portSTACK_TYPE ) 0x08080808; /* R8 */
156  pxTopOfStack--;
157  *pxTopOfStack = ( portSTACK_TYPE ) 0x07070707; /* R7 */
158  pxTopOfStack--;
159  *pxTopOfStack = ( portSTACK_TYPE ) 0x06060606; /* R6 */
160  pxTopOfStack--;
161  *pxTopOfStack = ( portSTACK_TYPE ) 0x05050505; /* R5 */
162  pxTopOfStack--;
163  *pxTopOfStack = ( portSTACK_TYPE ) 0x04040404; /* R4 */
164  pxTopOfStack--;
165  *pxTopOfStack = ( portSTACK_TYPE ) 0x03030303; /* R3 */
166  pxTopOfStack--;
167  *pxTopOfStack = ( portSTACK_TYPE ) 0x02020202; /* R2 */
168  pxTopOfStack--;
169  *pxTopOfStack = ( portSTACK_TYPE ) 0x01010101; /* R1 */
170  pxTopOfStack--;
171  *pxTopOfStack = ( portSTACK_TYPE ) pvParameters; /* R0 */
172  pxTopOfStack--;
173 
174  /* The last thing onto the stack is the status register, which is set for
175  system mode, with interrupts enabled. */
176  *pxTopOfStack = ( portSTACK_TYPE ) portINITIAL_SPSR;
177 
178  if( ( ( unsigned long ) pxCode & 0x01UL ) != 0x00UL )
179  {
180  /* We want the task to start in thumb mode. */
181  *pxTopOfStack |= portTHUMB_MODE_BIT;
182  }
183 
184  pxTopOfStack--;
185 
186  /* The code generated by the Keil compiler does not maintain separate
187  stack and frame pointers. The portENTER_CRITICAL macro cannot therefore
188  use the stack as per other ports. Instead a variable is used to keep
189  track of the critical section nesting. This variable has to be stored
190  as part of the task context and is initially set to zero. */
191  *pxTopOfStack = portNO_CRITICAL_SECTION_NESTING;
192 
193  return pxTopOfStack;
194 }

251 static void prvSetupTimerInterrupt( void )
252 {
253 unsigned portLONG ulCompareMatch;
254 
255  /* A 1ms tick does not require the use of the timer prescale. This is
256  defaulted to zero but can be used if necessary. */
257  T0PR = portPRESCALE_VALUE;
258 
259  /* Calculate the match value required for our wanted tick rate. */
260  ulCompareMatch = configCPU_CLOCK_HZ / configTICK_RATE_HZ;
261 
262  /* Protect against divide by zero. Using an if() statement still results
263  in a warning - hence the #if. */
264  #if portPRESCALE_VALUE != 0
265  {
266  ulCompareMatch /= ( portPRESCALE_VALUE + 1 );
267  }
268  #endif
269 
270  T0MR0 = ulCompareMatch;
271 
272  /* Generate tick with timer 0 compare match. */
273  T0MCR = portRESET_COUNT_ON_MATCH | portINTERRUPT_ON_MATCH;
274 
275  /* Setup the VIC for the timer. */
276  VICIntSelect &= ~( portTIMER_VIC_CHANNEL_BIT );
277  VICIntEnable |= portTIMER_VIC_CHANNEL_BIT;
278 
279  /* The ISR installed depends on whether the preemptive or cooperative
280  scheduler is being used. */
281  #if configUSE_PREEMPTION == 1
282  {
283  VICVectAddr0 = ( unsigned portLONG ) vPreemptiveTick;
284  }
285  #else
286  {
287  VICVectAddr0 = ( unsigned portLONG ) vNonPreemptiveTick;
288  }
289  #endif
290 
291  VICVectCntl0 = portTIMER_VIC_CHANNEL | portTIMER_VIC_ENABLE;
292 
293  /* Start the timer - interrupts are disabled when this function is called
294  so it is okay to do this here. */
295  T0TCR = portENABLE_TIMER;
296 }

299 void vPortEnterCritical( void )
300 {
301  /* Disable interrupts as per portDISABLE_INTERRUPTS(); */
302  __disable_irq();
303 
304  /* Now interrupts are disabled ulCriticalNesting can be accessed
305  directly. Increment ulCriticalNesting to keep a count of how many times
306  portENTER_CRITICAL() has been called. */
307  ulCriticalNesting++;
308 }

310 
311 void vPortExitCritical( void )
312 {
313  if( ulCriticalNesting > portNO_CRITICAL_NESTING )
314  {
315  /* Decrement the nesting count as we are leaving a critical section. */
316  ulCriticalNesting--;
317 
318  /* If the nesting level has reached zero then interrupts should be
319  re-enabled. */
320  if( ulCriticalNesting == portNO_CRITICAL_NESTING )
321  {
322  /* Enable interrupts as per portEXIT_CRITICAL(). */
323  __enable_irq();
324  }
325  }
326 }
