74 /* Write Cursor Image into Internal Cursor Image Buffer */
75 void Chip_LCD_Cursor_WriteImage(LPC_LCD_T *pLCD, uint8_t cursor_num, void *Image)
76 {
77  int i, j;
78  uint32_t *fifoptr, *crsr_ptr = (uint32_t *) Image;
79 
80  /* Check if Cursor Size was configured as 32x32 or 64x64*/
81  if (LCD_Cursor_Size == LCD_CURSOR_32x32) {
82  i = cursor_num * 64;
83  j = i + 64;
84  }
85  else {
86  i = 0;
87  j = 256;
88  }
89  fifoptr = IP_LCD_Cursor_GetImageBufferAddress(pLCD, 0);
90 
91  /* Copy Cursor Image content to FIFO */
92  for (; i < j; i++) {
93 
94  *fifoptr = *crsr_ptr;
95  crsr_ptr++;
96  fifoptr++;
97  }
98 }

100 /* Load LCD Palette */
101 void Chip_LCD_LoadPalette(LPC_LCD_T *pLCD, void *palette) {
102  LCD_PALETTE_ENTRY_T pal_entry, *ptr_pal_entry;
103  uint8_t i, *pal_ptr;
104 
105  ptr_pal_entry = &pal_entry;
106  pal_ptr = (uint8_t *) palette;
107 
108  /* 256 entry in the palette table */
109  for (i = 0; i < 256 / 2; i++) {
110  pal_entry.Bl = (*pal_ptr++) >> 3; /* blue first */
111  pal_entry.Gl = (*pal_ptr++) >> 3; /* get green */
112  pal_entry.Rl = (*pal_ptr++) >> 3; /* get red */
113  pal_ptr++; /* skip over the unused byte */
114  /* do the most significant halfword of the palette */
115  pal_entry.Bu = (*pal_ptr++) >> 3; /* blue first */
116  pal_entry.Gu = (*pal_ptr++) >> 3; /* get green */
117  pal_entry.Ru = (*pal_ptr++) >> 3; /* get red */
118  pal_ptr++; /* skip over the unused byte */
119 
120  IP_LCD_Color_LoadPalette(pLCD, (uint32_t *) &ptr_pal_entry, i);
121  }
122 }
