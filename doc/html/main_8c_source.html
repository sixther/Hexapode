57 void vTIMER0_Init()
58 {
59  uint32_t timerFreq;
60 
61  Chip_TIMER_Init(LPC_TIMER0);
62  /* on recupére la fréquence du périphérique */
63  timerFreq = Chip_Clock_GetPeripheralClockRate(SYSCTL_PCLK_TIMER0);
64  Chip_TIMER_Reset(LPC_TIMER0);
65  /* on active l'interruption sur match0 */
66  Chip_TIMER_MatchEnableInt(LPC_TIMER0, 0);
67  /* on défini la fréquence du timer */
68  Chip_TIMER_SetMatch(LPC_TIMER0, 0, (timerFreq / TICKRATE1_HZ));
69  /* on reset le timer quand match */
70  Chip_TIMER_ResetOnMatchEnable(LPC_TIMER0, 0);
71  /* activation du timer */
72  Chip_TIMER_Enable(LPC_TIMER0);
73  /* Enable timer interrupt */
74  NVIC_EnableIRQ(TIMER0_IRQn);
75  NVIC_ClearPendingIRQ(TIMER0_IRQn);
76 }
