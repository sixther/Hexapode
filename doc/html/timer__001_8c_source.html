50 /* Resets the timer terminal and prescale counts to 0 */
51 void IP_TIMER_Reset(IP_TIMER_001_T *pTimer)
52 {
53  uint32_t reg;
54 
55  /* Disable timer, set terminal count to non-0 */
56  reg = pTimer->TCR;
57  pTimer->TCR = 0;
58  pTimer->TC = 1;
59 
60  /* Reset timer counter */
61  pTimer->TCR = TIMER_RESET;
62 
63  /* Wait for terminal count to clear */
64  while (pTimer->TC != 0) {}
65 
66  /* Restore timer state */
67  pTimer->TCR = reg;
68 }
