54 /* Basic Ethernet interface initialization */
55 void Chip_ENET_Init(LPC_ENET_T *pENET)
56 {
57  volatile uint32_t i;
58 
59  /* Enable Ethenet Clock */
60  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_ENET);
61 #if !defined(CHIP_LPC175X_6X)
62  Chip_SYSCTL_PeriphReset(SYSCTL_RESET_ENET);
63 #endif
64 
65  /* Reset ethernet peripheral, small delay after reset */
66  Chip_ENET_Reset(pENET);
67  for (i = 0; i < 100; i++) {}
68 
69  IP_ENET_Init(pENET);
70 }

72 /* Find the divider index for a desired MII clock rate */
73 uint32_t Chip_ENET_FindMIIDiv(LPC_ENET_T *pENET, uint32_t clockRate)
74 {
75  uint32_t tmp, divIdx = 0;
76 
77  /* Find desired divider value */
78  tmp = Chip_Clock_GetENETClockRate() / clockRate;
79 
80  /* Determine divider index from desired divider */
81  for (divIdx = 0; divIdx < (sizeof(EnetClkDiv) / sizeof(EnetClkDiv[0])); divIdx++) {
82  /* Closest index, but not higher than desired rate */
83  if (EnetClkDiv[divIdx] >= tmp) {
84  return divIdx;
85  }
86  }
87 
88  /* Use maximum divider index */
89  return (sizeof(EnetClkDiv) / sizeof(EnetClkDiv[0])) - 1;
90 }
