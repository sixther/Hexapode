82 /* Setup system clocking */
83 STATIC void SystemSetupClocking(void)
84 {
85  /* CPU clock source starts with IRC */
86  Chip_Clock_SetMainPllSource(SYSCTL_PLLCLKSRC_IRC);
87  Chip_Clock_SetCPUClockSource(SYSCTL_CCLKSRC_SYSCLK);
88 
89  /* Enable main oscillator used for PLLs */
90  LPC_SYSCTL->SCS = SYSCTL_OSCEC;
91  while ((LPC_SYSCTL->SCS & SYSCTL_OSCSTAT) == 0) {}
92 
93  /* PLL0 clock source is 12MHz oscillator, PLL1 can only be the
94  main oscillator */
95  Chip_Clock_SetMainPllSource(SYSCTL_PLLCLKSRC_MAINOSC);
96 
97  /* Setup PLL0 for a 480MHz clock. It is divided by CPU Clock Divider to create CPU Clock.
98  Input clock rate (FIN) is main oscillator = 12MHz
99  FCCO is selected for PLL Output and it must be between 275 MHz to 550 MHz.
100  FCCO = (2 * M * FIN) / N = integer multiplier of CPU Clock (120MHz) = 480MHz
101  N = 1, M = 480 * 1/(2*12) = 20 */
102  Chip_Clock_SetupPLL(SYSCTL_MAIN_PLL, 19, 0);/* Multiply by 20, Divide by 1 */
103 
104  /* Enable PLL0 */
105  Chip_Clock_EnablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_ENABLE);
106 
107  /* Change the CPU Clock Divider setting for the operation with PLL0.
108  Divide value = (480/120) = 4 */
109  Chip_Clock_SetCPUClockDiv(3); /* pre-minus 1 */
110 
111  /* Change the USB Clock Divider setting for the operation with PLL0.
112  Divide value = (480/48) = 10 */
113  Chip_Clock_SetUSBClockDiv(9); /* pre-minus 1 */
114 
115  /* Wait for PLL0 to lock */
116  while (!Chip_Clock_IsMainPLLLocked()) {}
117 
118  /* Connect PLL0 */
119  Chip_Clock_EnablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_ENABLE | SYSCTL_PLL_CONNECT);
120 
121  /* Wait for PLL0 to be connected */
122  while (!Chip_Clock_IsMainPLLConnected()) {}
123 
124  /* Setup USB PLL1 for a 48MHz clock
125  Input clock rate (FIN) is main oscillator = 12MHz
126  PLL1 Output = USBCLK = 48MHz = FIN * MSEL, so MSEL = 4.
127  FCCO = USBCLK = USBCLK * 2 * P. It must be between 156 MHz to 320 MHz.
128  so P = 2 and FCCO = 48MHz * 2 * 2 = 192MHz */
129  Chip_Clock_SetupPLL(SYSCTL_USB_PLL, 3, 1); /* Multiply by 4, Divide by 2 */
130 
131 #if 0 /* Use PLL1 output as USB Clock Source */
132  /* Enable PLL1 */
133  Chip_Clock_EnablePLL(SYSCTL_USB_PLL, SYSCTL_PLL_ENABLE);
134 
135  /* Wait for PLL1 to lock */
136  while (!Chip_Clock_IsUSBPLLLocked()) {}
137 
138  /* Connect PLL1 */
139  Chip_Clock_EnablePLL(SYSCTL_USB_PLL, SYSCTL_PLL_ENABLE | SYSCTL_PLL_CONNECT);
140 
141  /* Wait for PLL1 to be connected */
142  while (!Chip_Clock_IsUSBPLLConnected()) {}
143 #endif
144 
145  /* Setup FLASH access to 5 clocks (120MHz clock) */
146  Chip_FMC_SetFLASHAccess(FLASHTIM_120MHZ_CPU);
147 }

150 STATIC void SystemSetupMuxing(void)
151 {
152  int i;
153 
154  /* Setup system level pin muxing */
155  for (i = 0; i < (sizeof(pinmuxing) / sizeof(pinmuxing[0])); i++) {
156  Chip_IOCON_PinMux(LPC_IOCON, pinmuxing[i].pingrp, pinmuxing[i].pinnum,
157  pinmuxing[i].pincfg, pinmuxing[i].funcnum);
158  }

175 
176 #if defined(__IAR_SYSTEMS_ICC__)

180 #elif defined(__CODE_RED)

188 #endif
